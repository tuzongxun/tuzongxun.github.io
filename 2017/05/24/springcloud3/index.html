springcloud

springcloudеҫ®жңҚеҠЎдёүпјҡEurekaжңҚеҠЎжІ»зҗҶд№ӢжіЁеҶҢжңҚеҠЎжҸҗдҫӣиҖ…еҸҠжңҚеҠЎзҡ„еҸ‘зҺ°е’Ңж¶Ҳиҙ№

иҜҙжҳҺ

еҪ“жңҚеҠЎжіЁеҶҢдёӯеҝғжҲҗеҠҹе»әз«Ӣд»ҘеҗҺпјҢе°ұйңҖиҰҒжңүжңҚеҠЎзҡ„жіЁеҶҢе’Ңж¶Ҳиҙ№пјҢеҗҰеҲҷиҝҷдёӘжңҚеҠЎжіЁеҶҢдёӯеҝғе°ұжІЎжңүдәҶеӯҳеңЁзҡ„ж„Ҹд№үпјҢиҖҢе®һйҷ…дёҠпјҢдёҖдёӘз®ҖеҚ•зҡ„жңҚеҠЎжіЁеҶҢд№ҹжҳҜйқһеёёз®ҖеҚ•зҡ„пјҢд»…д»…йңҖиҰҒе®һзҺ°еӣӣйғЁжӣІе°ұеҘҪгҖӮ

жңҚеҠЎжіЁеҶҢ

жӯҘйӘӨдёҖ

йҰ–е…ҲпјҢиҝҳжҳҜе»әз«ӢдёҖдёӘеҹәжң¬зҡ„spring bootзҡ„йЎ№зӣ®пјҢеҲӣе»әзҡ„ж—¶еҖҷйҖүжӢ©webпјҢз„¶еҗҺеңЁpom.xmlж–Үд»¶дёӯеҠ е…ҘEurekaзҡ„дҫқиө–пјҢе’ҢжңҚеҠЎжіЁеҶҢдёӯеҝғзҡ„mavenй…ҚзҪ®дёҖж ·пјҢиҝҷйҮҢе°ұдёҚеҶҚзҪ—еҲ—д»Јз ҒгҖӮ

жӯҘйӘӨдәҢ

з¬¬дәҢжӯҘпјҢе°ұжҳҜдҝ®ж”№application.propertiesж–Үд»¶пјҢжҢҮе®ҡз«ҜеҸЈе’ҢжңҚеҠЎеҗҚз§°д»ҘеҸҠжіЁеҶҢдёӯеҝғзҡ„ең°еқҖпјҡ

server.port=1001
spring.application.name=hello-service
eureka.client.serviceUrl.defaultZone=http://server1:1000/eureka/,http://server2:2000/eureka/

еӣ дёәдёҠдёҖжӯҘдёӯе®һзҺ°дәҶжіЁеҶҢдёӯеҝғзҡ„йӣҶзҫӨжЁЎејҸпјҢеӣ жӯӨиҝҷйҮҢеҸҜд»ҘжҢҮе®ҡдёӨдёӘпјҢеҰӮжһңжңүжӣҙеӨҡзҡ„жңҚеҠЎжіЁеҶҢдёӯеҝғпјҢиҝҳеҸҜд»ҘжҢҮе®ҡжӣҙеӨҡпјҢдёӯй—ҙд»ҘйҖ—еҸ·еҲҶйҡ”гҖӮ

жӯҘйӘӨдёү

з¬¬дёүжӯҘпјҢдҪҝз”ЁжіЁи§ЈејҖеҗҜиҮӘеҠЁиЈ…й…ҚпјҢдҪҝжңҚеҠЎиғҪеӨҹеҗ‘жіЁеҶҢдёӯеҝғжіЁеҶҢгҖҒеҗ‘жіЁеҶҢдёӯеҝғжңҚеҠЎз§ҹзәҰпјҲе®һйҷ…дёҠе°ұжҳҜе®ҡж—¶еҷЁжҢҒз»ӯеҸ‘йҖҒеҝғи·іе‘ҠиҜүжңҚеҠЎжіЁеҶҢдёӯеҝғиҮӘе·ұиҝҳжҙ»зқҖпјүгҖҒеҸ–ж¶Ҳз§ҹзәҰгҖҒжҹҘиҜўжңҚеҠЎеҲ—иЎЁзӯүпјҢиҝҷдёӘжіЁи§ЈжҳҜ@EnableDiscoveryClientпјҢж·»еҠ д»ҘеҗҺзҡ„д»Јз ҒеҰӮдёӢпјҡ

@EnableDiscoveryClient
@SpringBootApplication
public class EurekaService1Application {
    public static void main(String[] args) {
        SpringApplication.run(EurekaService1Application.class, args);
    }
}

жӯҘйӘӨеӣӣ

з¬¬еӣӣжӯҘпјҢжҲ‘д»¬иҜҙиҰҒиҝӣиЎҢжңҚеҠЎжіЁеҶҢпјҢеҪ“дёҠиҫ№дёҖеҲҮеҮҶеӨҮе·ҘдҪңеҒҡеҘҪд»ҘеҗҺпјҢиҮӘз„¶йңҖиҰҒжңүжңҚеҠЎжүҚиғҪжіЁеҶҢпјҢиҖҢиҝҷдёӘжңҚеҠЎе°ұжҳҜдёҖдёӘзңӢиө·жқҘеҫҲжҷ®йҖҡзҡ„controllerзұ»пјҢеҰӮеӣҫпјҡ

@RestController
public class HelloController {  
    @Autowired
    private DiscoveryClient client;
    @RequestMapping(value="/hello",method=RequestMethod.GET)
    public String sayHello(){
        ServiceInstance instance=client.getLocalServiceInstance();
        return "hello "+instance.getHost()+","+instance.getPort()+","+instance.getServiceId();
    }
}

иҝҷйҮҢзҡ„DiscoveryClientеҸҜд»ҘиҺ·еҸ–жіЁеҶҢеҲ°жңҚеҠЎжіЁеҶҢдёӯеҝғзҡ„жңҚеҠЎзҡ„зӣёе…ідҝЎжҒҜпјҢе№¶дёҚжҳҜеҝ…йЎ»жіЁе…Ҙзҡ„пјҢиҝҷйҮҢе®Ңе…ЁеҸҜд»ҘеҶҷжҲҗеҰӮдёӢжүҖзӨәзҡ„д»Јз Ғпјҡ

@RestController
public class HelloController {  
    @RequestMapping(value="/hello",method=RequestMethod.GET)
    public String sayHello(){
        return "hello";
    }
}

еҪ“з„¶дәҶпјҢеӣ дёәеҸӘжҳҜдәҶи§ЈжңҚеҠЎжіЁеҶҢзҡ„иҝҮзЁӢпјҢжүҖд»ҘиҝҷйҮҢд»…д»…иҝҷд№Ҳз®ҖеҚ•пјҢе®һйҷ…зҡ„дёҡеҠЎйҖ»иҫ‘иҮӘз„¶дёҚдјҡжҳҜиҝҷж ·гҖӮ

йӮЈд№ҲйҖҡиҝҮдёҠиҫ№еӣӣжӯҘпјҢдёҖдёӘиғҪеӨҹжіЁеҶҢеҲ°жңҚеҠЎжіЁеҶҢдёӯеҝғжҸҗдҫӣжңҚеҠЎзҡ„жңҚеҠЎе°ұе®ҢжҲҗдәҶпјҢеҗҜеҠЁиҝҷдёӘжңҚеҠЎе°ұдјҡиў«жіЁеҶҢеҲ°жңҚеҠЎжіЁеҶҢдёӯеҝғпјҢжҲ‘д»¬еҸҜд»ҘйҖҡиҝҮжңҚеҠЎжіЁеҶҢдёӯеҝғзҡ„йЎөйқўжқҘжҹҘзңӢпјҡ

жҲҗеҠҹзҡ„еҲӣе»әдәҶдёҖдёӘжңҚеҠЎе№¶жіЁеҶҢеҲ°жіЁеҶҢдёӯеҝғпјҢеҗҢж ·зҡ„пјҢдёәдәҶй«ҳеҸҜз”ЁпјҢжҲ‘д»¬еҸҜд»ҘеҶҚе»әдёҖдёӘе…·жңүеҗҢж ·еҠҹиғҪзҡ„жңҚеҠЎпјҢиҖҢиҝҷдёӘиҝҮзЁӢе°ұжӣҙеҠ зҡ„з®ҖеҚ•пјҢеҸӘйңҖиҰҒжӣҙж”№дёҖдёӢз«ҜеҸЈеҸ·пјҢд№ҹе°ұжҳҜserver.portпјҢз„¶еҗҺйҮҚж–°еҗҜеҠЁе°ұеҸҜд»ҘпјҢеҗҜеҠЁеҗҺдјҡзңӢеҲ°дёҠиҫ№зҡ„йЎөйқўеҰӮдёӢпјҲеҪ“з„¶дәҶпјҢиҝҷйҮҢд№ҹжҳҜдёәдәҶжөӢиҜ•ж–№дҫҝпјҢе®һйҷ…жғ…еҶөд№ҹеә”иҜҘжҳҜеңЁдёӨеҸ°жңәеҷЁдёҠиҝҗиЎҢзҡ„дёӨеҘ—д»Јз ҒжүҚеҜ№пјүпјҡ

иҝҷйҮҢжҲ‘д»¬дјҡзңӢеҲ°е·Із»ҸжңүдәҶдёӨдёӘеҗҢеә”з”ЁеҗҚзҡ„жңҚеҠЎпјҢз«ҜеҸЈеҲҶеҲ«жҳҜ1001е’Ң2001гҖӮ
иҖҢеңЁдёҠиҝ°йЎөйқўдёӯиҝҳйңҖиҰҒжіЁж„Ҹзҡ„жҳҜйӮЈдёҖж®өйҶ’зӣ®зҡ„зәўиүІжҸҗзӨәпјҢе®ғзҡ„ж„ҸжҖқе°ұжҳҜиҜҙејҖеҗҜдәҶиҮӘжҲ‘дҝқжҠӨжңәеҲ¶пјҢд№ҹе°ұжҳҜиҜҙеҪ“жңҚеҠЎеӨұж•ҲеҗҺпјҢжңүеҸҜиғҪжіЁеҶҢдёӯеҝғиҝҳдјҡдҝқз•ҷдёҖж®өж—¶й—ҙиҜҘжңҚеҠЎзҡ„дҝЎжҒҜпјҢиҝҷж ·е°ұеҸҜиғҪеҜјиҮҙжңҚеҠЎж¶Ҳиҙ№ж–№иҺ·еҸ–жңҚеҠЎеҲ—иЎЁзҡ„ж—¶еҖҷдҫқз„¶иғҪеӨҹиҺ·еҸ–еҲ°иҝҷдёӘжңҚеҠЎпјҢдҪҶжҳҜе®һйҷ…дёҠеҗ‘иҜҘжңҚеҠЎеҸ‘йҖҒиҜ·жұӮзҡ„ж—¶еҖҷеҸҲж— жі•жҲҗеҠҹиҜ·жұӮпјҢйӮЈд№ҲеӯҰд№ зҡ„иҝҮзЁӢдёӯеҸҜд»ҘдҪҝз”Ёд»ҘдёӢй…ҚзҪ®е…ій—ӯиҮӘжҲ‘дҝқжҠӨжңәеҲ¶пјҢз„¶еҗҺеҰӮжһңеҶҚи®ҝй—®жіЁеҶҢдёӯеҝғзҡ„йЎөйқўдҫҝдёҚдјҡеҶҚеҮәзҺ°иҝҷдёӘжҸҗзӨәпјҢд№ҹдёҚдјҡеҮәзҺ°йӮЈз§ҚжңҚеҠЎеӨұж•ҲеҗҺиҝҳдҝқз•ҷеңЁжіЁеҶҢдёӯеҝғзҡ„жғ…еҶөгҖӮ

#и®ҫзҪ®жҳҜеҗҰејҖеҗҜиҮӘжҲ‘дҝқжҠӨжңәеҲ¶пјҢй»ҳи®ӨжҳҜtrue
eureka.server.enable-self-preservation=false

жңҚеҠЎеҸ‘зҺ°

жңҚеҠЎжіЁеҶҢеҘҪд»ҘеҗҺпјҢдёӢиҫ№е°ұжҳҜжңҚеҠЎзҡ„еҸ‘зҺ°е’Ңж¶Ҳиҙ№пјҢйҖҡеёёжңүribbonе’ҢfeignдёӨз§Қж–№ејҸпјҢиҖҢfeignе®һйҷ…дёҠд№ҹжҳҜд»ҘribbonдёәеҹәзЎҖзҡ„пјҢиҝҷдёӨз§ҚеҗҺиҫ№йғҪиҝҳйңҖиҰҒе…·дҪ“иҜҰз»Ҷзҡ„еӯҰд№ пјҢжүҖд»ҘиҝҷйҮҢеҸӘз®ҖеҚ•зҡ„жј”зӨәдёҖдёӢж¶Ҳиҙ№зҡ„иҝҮзЁӢпјҢдёҚеҒҡеӨӘиҜҰз»Ҷзҡ„и§ЈйҮҠгҖӮribbonеҸҜд»Ҙе®һзҺ°иҙҹиҪҪеқҮиЎЎпјҢиҝҷйҮҢдҪҝз”ЁжңҖз®ҖеҚ•зҡ„жЁЎжқҝе®һзҺ°гҖӮ
йҰ–е…ҲиҝҳжҳҜе»әз«ӢдёҖдёӘеҹәзЎҖspring bootйЎ№зӣ®пјҢз„¶еҗҺpom.xmlдёӯеҠ е…Ҙeurekaе’Ңribbonзҡ„дҫқиө–д»ҘеҸҠeurekaзҡ„зүҲжң¬дҝЎжҒҜпјҡ

    <groupId>org.springframework.cloud</groupId>
    <artifactId>spring-cloud-starter-eureka</artifactId>
</dependency>

з„¶еҗҺapplication.propertiesж–Үд»¶еҶ…е®№еҰӮдёӢпјҡ

еҶҚз„¶еҗҺе°ұжҳҜеҲӣе»әйЎ№зӣ®ж—¶з”ҹжҲҗзҡ„зұ»дёӯеҠ е…Ҙ@EnableDiscoveryClientеҸӮж•°ејҖеҗҜиҮӘеҠЁиЈ…й…ҚпјҢд»ҘеҸҠжіЁе…ҘжңҚеҠЎж¶Ҳиҙ№жЁЎжқҝе№¶ејҖеҗҜиҙҹиҪҪеқҮиЎЎпјҡ

@Bean
@LoadBalanced
RestTemplate restTemplate(){
    return new RestTemplate();
}

д№ӢеҗҺзҡ„иҝҷдёӘжӯҘйӘӨе®һйҷ…дёҚжҳҜеҝ…иҰҒзҡ„пјҢеҸӘжҳҜдёәдәҶиғҪеӨҹжӣҙзӣҙи§Ӯзҡ„еңЁйЎөйқўдёӯзңӢеҲ°з»“жһңпјҢжүҖд»ҘеҠ е…ҘдёҖдёӘcontrollerзұ»пјҢиҫ“еҮәжңҚеҠЎзҡ„дёҖдәӣеҹәжң¬дҝЎжҒҜпјҡ

@RestController
public class ConsumerController {    
    @Autowired
    RestTemplate restTemplate;
    @RequestMapping(value="/hello",method=RequestMethod.GET)
    public String hello(){
        return restTemplate.getForEntity("http://HELLO-SERVICE/hello", String.class).getBody();
    }
}
